Run script

  <script>
    $(".jsa-wrapper").slideAnything();
  </script>

Options

  <script>
    $(".jsa-wrapper").slideAnything({
      timerSpeed: 2000, // time to wait for next slide
      toggleSpeed: 300, // time for toggleClass animation
    });
  </script>

  Note: always keep timerSpeed higher than toggleSpeed
